% Force certain characters to use Latin (ALchar) font instead of CJK (JAchar) font.
% By default, luatexja classifies these as JAchar, rendering them with the CJK
% font (e.g. Noto Sans CJK SC) which may lack the glyphs. This override ensures
% they use the Latin font fallback chain (which includes Noto Color Emoji).
%
% luatexja supports custom character ranges 1-217. We use range 217 for:
% - U+2018..U+201D: smart quotes (use Latin font, not CJK)
% - U+2600..U+27BF: Misc Symbols + Dingbats (✅ U+2705, ❌ U+274C → Noto Color Emoji)
\ltjdefcharrange{217}{"2018-"201D,"2600-"27BF}
\ltjsetparameter{jacharrange={-217}}

% Override Highlighting environment: remove breakanywhere to prevent
% "TeX capacity exceeded [dest_names_size=131072]" in LuaLaTeX.
% The base listings.tex uses breakanywhere which creates a potential break point
% at every character, exhausting LuaTeX's 131072 destination name limit in
% code-heavy documents. We keep breaklines (wraps at spaces) instead.
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{breaklines,breakautoindent=false,breaknonspaceingroup,frame=single,framesep=8pt,rulecolor=\color[HTML]{aaaaaa},commandchars=\\\{\}}

% Define \pandocbounded for pandoc >= 3.2.1 compatibility.
% The custom default.latex template (from dotfiles-public) predates this macro.
% Official definition from jgm/pandoc-templates common.latex:
\makeatletter
\@ifundefined{pandocbounded}{%
  \newsavebox\pandoc@box
  \newcommand*\pandocbounded[1]{% scales image to fit in text height/width
    \sbox\pandoc@box{#1}%
    \Gscale@div\@tempa{\textheight}{\dimexpr\ht\pandoc@box+\dp\pandoc@box\relax}%
    \Gscale@div\@tempb{\linewidth}{\wd\pandoc@box}%
    \ifdim\@tempb\p@<\@tempa\p@\let\@tempa\@tempb\fi
    \ifdim\@tempa\p@<\p@\scalebox{\@tempa}{\usebox\pandoc@box}%
    \else\usebox{\pandoc@box}%
    \fi
  }%
}{}
\makeatother
