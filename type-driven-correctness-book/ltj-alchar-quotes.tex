% Force certain characters to use Latin (ALchar) font instead of CJK (JAchar) font.
% By default, luatexja classifies these as JAchar, rendering them with the CJK
% font (e.g. Noto Sans CJK SC) which may lack the glyphs. This override ensures
% they use the Latin font fallback chain (which includes Noto Color Emoji).
%
% luatexja supports custom character ranges 1-217. We use range 217 for:
% - U+2018..U+201D: smart quotes (use Latin font, not CJK)
% - U+2600..U+27BF: Misc Symbols + Dingbats (✅ U+2705, ❌ U+274C → Noto Color Emoji)
\ltjdefcharrange{217}{"2018-"201D,"2600-"27BF}
\ltjsetparameter{jacharrange={-217}}
